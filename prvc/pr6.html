    #include<stdio.h>
    #include<conio.h>
    #include<math.h>
        
    int main ()
    {
    float pi=3.141593;
    //задаем радиусы
    float R_sph=2,R_cyl=2,R_con=2;
    float H_cyl=5,H_con=5; 
    //задаем высоты 
    float L_con;

    S_sph = 4*pi*R_sph*R_sph;

    // Вычисляем параметры цилиндра 
    S_cyl = 2*pi*R_cyl*(H_cyl+R_cyl); 
    V_cyl = pi*R_cyl*R_cyl*H_cyl;
    printf("Cylinder:s=%f v=%f\n", S_cyl,V_cyl); //печать на экран
        
    // Вычисляем параметры конуса 
    L_con = hypot(H_con,R_con); 
    S_con = pi*R_con*(L_con+R_con); 
    V_con = 1./3*pi*R_con*R_con*H_con; 
    printf("Cone: s=%f v=%f\n",S_con,V_con); //печать на экран 
        
    getch(); 
    return 0;
    }

    #include<stdio.h>
    #include<conio.h>
    #include<math.h>
        
    int main()
    {
    int x1,y1,x2,y2,x3,y3;
    float a,b,c,S,p;
        
    // Задание координат вершин 
    x1=0;	y1=0;
    x2=4;	y2=0;
    x3=0;	y3=3;
        
    // Вычисление сторон треугольника по формуле Пифагора
    a=hypot(x2-x1,y2-y1); 	//х2-х1 и у2-у1 – длины катетов
    b=hypot(x3-x2,y3-y2) ; 	//хЗ-х2 и у3-у2 – длины катетов
    c=hypot(x1-x3,y1-y3); 	//x1-х3 и y1-у3 – длины катетов
        
    // Вычисление площади треугольника по формуле Герона 
    p=(a+b+c)/2;
    S=sqrt (p*(p-a) * (p-b) * (p-c)) ;
        
    printf("%f",S); //вывод результата на экран 
    getch();
    return 0;
    }

    Ответ: 6.000000




    Задача 3. Написать программу для вычисления объема и площади поверхности кольца. Размеры кольца задать самостоятельно.


    Фрагмент кода:



    #include<stdio.h>

    #define PI 3.14159
        
    int main() {
    // Размеры кольца (радиус внешнего и внутреннего кругов)
    double outer_radius = 5.0; // Радиус внешнего круга
    double inner_radius = 3.0; // Радиус внутреннего круга
        
    // Вычисление площади поверхности кольца
    double area = PI * (outer_radius * outer_radius - inner_radius * inner_radius);
        
    // Вычисление объема кольца
    double volume = PI * (outer_radius * outer_radius - inner_radius * inner_radius) * 1; // Толщина кольца задаем как 1 
        
    // Вывод результатов
    printf("Площадь поверхности кольца: %.2f\n", area);
    printf("Объем кольца: %.2f\n", volume);
        
    return 0;
    }

    Ответ:



    Площадь поверхности кольца: 50.27


    Объем кольца: 100.53
